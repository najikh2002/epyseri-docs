    Usage

    Guide for using this package

    Init

    PARAMETERS

    Name Type Description Default
    base_url str Base URL of the Trident Trace service required
    auth_token str Authentication token, default is None opsional
    app_name str Application name, default is None opsional

    RETURNS

    Name Type Description Default
    ttc str Base URL of the Trident Trace service required

    USAGE

    from epyseri import TridentTraceClient

ttc = TridentTraceClient(
    base_url="string",
    app_name="string",
    auth_token="string"
)

print(ttc.service_domain_id, ttc.service_domain_slug)

    Inference

    GET /api/v1/inference - List Inference

    PARAMETERS

    Parameter Type Description Default
    business_domain str The business domain to filter the inferences.
    limit int (optional) The maximum number of results to return. Default is None. None
    offset int (optional) The result offset. Default is None. None
    service_domain_id int (optional) The service domain ID to filter the inferences. Default is None. None
    service_domain_slug str (optional) The service domain slug to filter the inferences. Default is None. None
    ai_domain str (optional) The AI domain to filter the inferences. Default is None. None
    ai_subdomain str (optional) The AI subdomain to filter the inferences. Default is None. None
    user str (optional) The user to filter the inferences. Default is None. None
    group_id str (optional) The group ID to filter the inferences. Default is None. None
    order_by str (optional) The field to order the inferences. Default is None. None

    RETURNS

    Returns Type Description Default
    Any A list of inferences that match the specified parameters.

    USAGE

    get_list_inference_res = ttc.get_list_inference("string")

print(get_list_inference_res)

    POST /api/v1/inference - Create Inference

    PARAMETERS

    Parameter Type Description Default
    data_input str or dict Input data for the inference.
    data_output str or dict Output data for the inference.
    inference_time int Time of the inference.
    ai_domain str AI domain of the inference.
    ai_subdomain str AI subdomain of the inference.
    model_version str Version of the model used for the inference.
    business_domain str Business domain of the inference.
    user str User who performed the inference.
    image np.ndarray or str or None (optional) Image data used for the inference. Defaults to None. None
    group_id str or None (optional) Group ID of the inference. Defaults to None. None
    service_domain_id int or None (optional) Service domain ID of the inference. Defaults to None. None
    conf_threshold float or None (optional) Confidence threshold for the inference. Defaults to None. None

    RETURNS

    Returns Type Description Default
    dict or str Response data.

    USAGE

    import cv2

image = cv2.imread("./image.jpeg")

post_inference_res = ttc.post_inference(
        data_input = "string",
        data_output = "string",
        inference_time = 0,
        business_domain = "string",
        ai_domain = "string",
        ai_subdomain = "string",
        model_version = "string",
        group_id = "string",
        user = "test",
        image = image,
)
print(post_inference_res)

    Feedback

    POST /api/v1/inference/{inference_uuid} - Create Feedback

    PARAMETERS

    Parameter Type Description Default
    inference_uuid int Identifier of the associated inference.
    user_reaction bool User's reaction to the inference.
    user_comment str or dict User's comment on the inference.
    user str (optional) User who created the feedback. Default is None. None

    RETURNS

    Returns Type Description Default
    Any Response data.

    USAGE

    create_feedback_res = ttc.post_feedback(
    inference_uuid="74642c5b-c4ad-49f8-b1da-2d482bb424f8", 
    user_comment={
        "id": 22,
        "user": "nama",
        "message": "Test "
    }, 
    user_reaction=True
)

print(create_feedback_res)

    PUT /api/v1/inference/{inference_uuid} - Update Feedback

    PARAMETERS

    Parameter Type Description Default
    uuid str Identifier of the feedback to be updated.
    inference_uuid str Identifier of the associated inference.
    user_reaction bool User's updated reaction to the inference.
    user_comment str or dict User's updated comment on the inference.
    user str (optional) User who is updating the feedback. None

    RETURNS

    Returns Type Description Default
    Any Response data from the update_feedback method of the CoreAPI object.

    USAGE

    delete_inference_res = ttc.delete_inference("5c462e22-dad0-46eb-9af8-fb7f34a6cf06")

print(delete_inference_res)
